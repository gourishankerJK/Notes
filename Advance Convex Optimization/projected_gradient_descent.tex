\section{Projected Gradient Descent}
\begin{definition}[Projection onto a convex set]\label{def:projection}
    Let $\mathbb{X} \subset \mathbb{R}^n$ be nonempty, closed, and convex.
    The projection operator $\Pi_{\mathbb{X}} : \mathbb{R}^n \to \mathbb{X}$ is defined by
    \[
        \Pi_{\mathbb{X}}(\mathbf{z})
        :=
        \arg\min_{\mathbf{x} \in \mathbb{X}}
        \|\mathbf{x} - \mathbf{z}\|_2.
    \]
\end{definition}
\begin{proposition}[Properties of the projection]\label{prop:projection}
    Let $\mathbb{X} \subset \mathbb{R}^n$ be closed and convex.
    Then:
    \begin{enumerate}
        \item $\Pi_{\mathbb{X}}$ is well-defined and single-valued.
        \item $\Pi_{\mathbb{X}}$ is nonexpansive:
              \[
                  \|\Pi_{\mathbb{X}}(\mathbf{x}) - \Pi_{\mathbb{X}}(\mathbf{y})\|_2
                  \le
                  \|\mathbf{x} - \mathbf{y}\|_2.
              \]
        \item (Projection inequality)
              \[
                  \langle \mathbf{z} - \Pi_{\mathbb{X}}(\mathbf{z}),
                  \mathbf{x} - \Pi_{\mathbb{X}}(\mathbf{z}) \rangle \le 0,
                  \quad \forall \mathbf{x} \in \mathbb{X}.
              \]
    \end{enumerate}
\end{proposition}
\begin{proof}[Proof of (1): Existence]
    Let $\mathcal{C} \subset \mathbb{R}^n$ be nonempty, closed, and convex, and fix
    $\mathbf{x} \in \mathbb{R}^n$.
    Choose any point $\mathbf{x}_0 \in \mathcal{C}$ and consider the problem:
    \[
        \min_{\mathbf{z} \in \mathcal{C}} \frac12 \|\mathbf{z} - \mathbf{x}\|_2^2.
    \]

    Set
    \[
        R := \|\mathbf{x} - \mathbf{x}_0\|_2
        \quad \text{and} \quad
        \mathcal{K} := \mathcal{C} \cap \overline{B}(\mathbf{x}, R),
    \]
    where $\overline{B}(\mathbf{x}, R)$ denotes the closed ball centered at
    $\mathbf{x}$ with radius $R$.

    For any $\mathbf{z} \in \mathcal{C}$ such that
    $\mathbf{z} \notin \mathcal{K}$, we must have $\|\mathbf{z} - \mathbf{x}\|_2 > R$.
    Substituting $R$, this implies:
    \[
        \frac12 \|\mathbf{z} - \mathbf{x}\|_2^2
        >
        \frac12 \|\mathbf{x}_0 - \mathbf{x}\|_2^2.
    \]
    Since $\mathbf{x}_0 \in \mathcal{K}$ achieves a strictly lower objective value,
    no point outside $\mathcal{K}$ can be a minimizer.
    Therefore,
    \[
        \arg\min_{\mathbf{z} \in \mathcal{C}}
        \frac12 \|\mathbf{z} - \mathbf{x}\|_2^2
        =
        \arg\min_{\mathbf{z} \in \mathcal{K}}
        \frac12 \|\mathbf{z} - \mathbf{x}\|_2^2.
    \]

    The set $\mathcal{K}$ is the intersection of a closed set $\mathcal{C}$ and a closed, bounded ball,
    so $\mathcal{K}$ is closed and bounded (hence compact).
    Since the function
    \[
        \mathbf{z} \mapsto \frac12 \|\mathbf{z} - \mathbf{x}\|_2^2
    \]
    is continuous, it attains its minimum on the compact set $\mathcal{K}$ by the
    \emph{Weierstrass extreme value theorem}.
    Thus, a minimizer exists.\\

\end{proof}

\begin{algorithmblock}[title=Projected Gradient Descent]
    \label{alg:pgd}
    \textbf{Input:} $f$, $C$, $\gamma$, $\mathbf{x}_0$ \\
    \textbf{for} $k = 0,1,2,\dots$ \textbf{do} \\
    \hspace*{2em} $\mathbf{x}_{k+1} = \Pi_C\bigl(\mathbf{x}_k - \gamma \nabla f(\mathbf{x}_k)\bigr)$ \\
    \textbf{end for}
\end{algorithmblock}
\begin{remark}\label{rem:pgd}
    \begin{itemize}
        \item If $\mathbb{X} = \mathbb{R}^n$, Algorithm~\ref{alg:pgd} reduces to
              standard gradient descent.
        \item Constraints are enforced via projection rather than penalties.
        \item The analysis parallels unconstrained gradient descent using
              Fej\'er monotonicity.
    \end{itemize}
\end{remark}
\begin{theorem}[Theorem A]\label{thm:pgd-main}
    Let $C \subset \mathbb{R}^n$ be nonempty, closed, and convex, and let
    $f$ be differentiable on an open set containing $C$.
    Assume that the problem
    \[
        \min_{\mathbf{x} \in C} f(\mathbf{x})
    \]
    is solvable.

    Let $\{\mathbf{x}_k\}$ be the sequence generated by projected gradient descent.
    Then
    \[
        f(\mathbf{x}_k) - \min_{\mathbf{x} \in C} f(\mathbf{x})
        \le
        \mathcal{O}\!\left(\frac{1}{k}\right),
    \]
    and
    \[
        \mathbf{x}_k \to \mathbf{x}^\ast,
        \qquad
        \mathbf{x}^\ast \in \arg\min_{\mathbf{x} \in C} f(\mathbf{x}).
    \]
\end{theorem}
\begin{proposition}[Optimality condition]\label{prop:optimality-C}
    Let $C \subset \mathbb{R}^n$ be nonempty, closed, and convex, and let
    $f$ be differentiable on an open set containing $C$.
    For a point $\mathbf{x}^\ast \in C$, the following statements are equivalent:
    \begin{enumerate}
        \item \[\mathbf{x}^\ast \in \arg\min_{\mathbf{x} \in C} f(\mathbf{x})\]
        \item
              \[
                  \mathbf{x}^\ast
                  =
                  \Pi_C\bigl(\mathbf{x}^\ast - \gamma \nabla f(\mathbf{x}^\ast)\bigr),
                  \quad \forall \gamma > 0.
              \]
        \item
              \[
                  \langle \mathbf{x} - \mathbf{x}^\ast,
                  \nabla f(\mathbf{x}^\ast) \rangle
                  \ge 0,
                  \quad \forall \mathbf{x} \in C.
              \]
    \end{enumerate}
\end{proposition}
\begin{proposition}[Acuteness property of projection]\label{prop:acuteness}
    Let $C \subset \mathbb{R}^n$ be nonempty, closed, and convex.
    For $\mathbf{x}_0 \in \mathbb{R}^n$ and $\mathbf{p} \in C$, the following
    statements are equivalent:
    \begin{enumerate}
        \item \[\mathbf{p} = \Pi_C(\mathbf{x}_0)\]
        \item
              \[
                  \langle \mathbf{x}_0 - \mathbf{p}, \mathbf{x} - \mathbf{p} \rangle \le 0,
                  \quad \forall \mathbf{x} \in C.
              \]
    \end{enumerate}
\end{proposition}

\begin{remark}
    The inequality states that the angle between the vectors
    $\mathbf{x}_0 - \mathbf{p}$ and $\mathbf{x} - \mathbf{p}$ is obtuse or right
    for every $\mathbf{x} \in C$.
    Geometrically, this means that the segment joining $\mathbf{x}_0$ to its
    projection $\mathbf{p}$ forms a supporting hyperplane to the set $C$ at
    $\mathbf{p}$.
\end{remark}
